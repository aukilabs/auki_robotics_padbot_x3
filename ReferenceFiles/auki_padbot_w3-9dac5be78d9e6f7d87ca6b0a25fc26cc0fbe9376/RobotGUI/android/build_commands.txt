# Build Commands for RobotGUI Android App

# Build and install the release version (recommended for deployment to robot)
./gradlew assembleRelease && adb install -r ./app/build/outputs/apk/release/X3-Prep.apk

# Location of the release APK
./app/build/outputs/apk/release/X3-Prep.apk

# Launch the app (after installation)
adb shell monkey -p com.x3prep -c android.intent.category.LAUNCHER 1

# Other useful commands:
# Clean build
./gradlew clean

# Build debug version (for development only)
./gradlew assembleDebug

# Install debug version (for development only)
adb install -r app/build/outputs/apk/debug/app-debug.apk
